comissionamento_validacao.sql 
-- Validação de pagamentos de comissionamento
-- Cruza dados de emissão com regras de negócio

WITH base_comissao AS (
    SELECT
        c.id_corretor,
        c.nome_corretor,
        p.nr_proposta,
        p.cd_produto,
        p.vl_premio,
        p.vl_comissao,
        p.dt_emissao,
        r.pct_comissao AS pct_esperado,
        ROUND(p.vl_premio * r.pct_comissao / 100, 2) AS vl_comissao_esperado
    FROM dbo.fato_producao p
    INNER JOIN dbo.dim_corretor c
        ON p.id_corretor = c.id_corretor
    INNER JOIN dbo.regra_comissao r
        ON p.cd_produto = r.cd_produto
        AND p.dt_emissao BETWEEN r.dt_inicio AND r.dt_fim
    WHERE p.dt_emissao >= '2024-01-01'
)
SELECT
    id_corretor,
    nome_corretor,
    COUNT(*) AS qtd_propostas,
    SUM(vl_comissao) AS total_pago,
    SUM(vl_comissao_esperado) AS total_esperado,
    SUM(vl_comissao) - SUM(vl_comissao_esperado) AS diferenca
FROM base_comissao
GROUP BY id_corretor, nome_corretor
HAVING ABS(SUM(vl_comissao) - SUM(vl_comissao_esperado)) > 0.01
ORDER BY diferenca DESC;

Query de validação que cruza produção com regras de comissionamento, identificando divergências entre valores pagos e esperados por corretor.

etl_carga_incremental.sql 
-- Procedure de carga incremental para fato de sinistros
-- Utiliza merge para upsert e controle de watermark

CREATE PROCEDURE dbo.sp_carga_fato_sinistro
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @ultima_carga DATETIME;

    SELECT @ultima_carga = MAX(dt_ultima_carga)
    FROM dbo.controle_etl
    WHERE nm_processo = 'FATO_SINISTRO';

    MERGE dbo.fato_sinistro AS target
    USING (
        SELECT
            s.id_sinistro,
            s.id_apolice,
            s.cd_produto,
            s.vl_sinistro,
            s.dt_aviso,
            s.dt_pagamento,
            s.cd_status,
            GETDATE() AS dt_carga
        FROM staging.sinistros s
        WHERE s.dt_alteracao > @ultima_carga
    ) AS source
    ON target.id_sinistro = source.id_sinistro
    WHEN MATCHED THEN
        UPDATE SET
            target.vl_sinistro = source.vl_sinistro,
            target.cd_status   = source.cd_status,
            target.dt_carga    = source.dt_carga
    WHEN NOT MATCHED THEN
        INSERT (id_sinistro, id_apolice, cd_produto,
                vl_sinistro, dt_aviso, dt_pagamento,
                cd_status, dt_carga)
        VALUES (source.id_sinistro, source.id_apolice,
                source.cd_produto, source.vl_sinistro,
                source.dt_aviso, source.dt_pagamento,
                source.cd_status, source.dt_carga);

    UPDATE dbo.controle_etl
    SET dt_ultima_carga = GETDATE()
    WHERE nm_processo = 'FATO_SINISTRO';
END;

Procedure de carga incremental usando MERGE para upsert na tabela fato de sinistros, com controle de watermark para rastreabilidade.

etl_pipeline.py 
# Pipeline ETL para processamento de dados de produção
# Extrai do banco, transforma e carrega no data warehouse

import pandas as pd
from sqlalchemy import create_engine
from datetime import datetime

def extrair_producao(engine, data_inicio):
    """Extrai dados de produção a partir da data informada."""
    query = f"""
        SELECT id_proposta, cd_produto, vl_premio,
               dt_emissao, cd_corretor, cd_status
        FROM producao.propostas
        WHERE dt_emissao >= '{data_inicio}'
          AND cd_status = 'EMI'
    """
    return pd.read_sql(query, engine)

def transformar_dados(df):
    """Aplica regras de negócio e limpeza."""
    df['dt_emissao'] = pd.to_datetime(df['dt_emissao'])
    df['ano_mes'] = df['dt_emissao'].dt.to_period('M')
    df['faixa_premio'] = pd.cut(
        df['vl_premio'],
        bins=[0, 1000, 5000, 20000, float('inf')],
        labels=['Baixo', 'Médio', 'Alto', 'Premium']
    )
    df = df.dropna(subset=['cd_corretor'])
    return df

def carregar_dw(df, engine_dw):
    """Carrega dados transformados no DW."""
    df.to_sql(
        name='fato_producao',
        con=engine_dw,
        if_exists='append',
        index=False,
        chunksize=5000
    )
    print(f"[{datetime.now()}] {len(df)} registros carregados.")

if __name__ == '__main__':
    src = create_engine('mssql+pyodbc://server/db_prod')
    dwh = create_engine('mssql+pyodbc://server/dw')

    dados = extrair_producao(src, '2024-01-01')
    dados = transformar_dados(dados)
    carregar_dw(dados, dwh)

Pipeline ETL em Python usando Pandas e SQLAlchemy para extração, transformação e carga de dados de produção no data warehouse.

qualidade_dados.py 
# Verificação automatizada de qualidade de dados
# Gera relatório de anomalias e inconsistências

import pandas as pd
import json
from datetime import datetime

class QualidadeDados:
    def __init__(self, df, nome_tabela):
        self.df = df
        self.nome = nome_tabela
        self.problemas = []

    def verificar_nulos(self, colunas_obrigatorias):
        for col in colunas_obrigatorias:
            nulos = self.df[col].isnull().sum()
            if nulos > 0:
                self.problemas.append({
                    'tipo': 'NULO',
                    'coluna': col,
                    'qtd': int(nulos),
                    'pct': round(nulos / len(self.df) * 100, 2)
                })

    def verificar_duplicatas(self, chaves):
        dupl = self.df.duplicated(subset=chaves).sum()
        if dupl > 0:
            self.problemas.append({
                'tipo': 'DUPLICATA',
                'chaves': chaves,
                'qtd': int(dupl)
            })

    def gerar_relatorio(self):
        relatorio = {
            'tabela': self.nome,
            'dt_verificacao': datetime.now().isoformat(),
            'total_registros': len(self.df),
            'problemas': self.problemas,
            'status': 'OK' if not self.problemas else 'ATENÇÃO'
        }
        return json.dumps(relatorio, indent=2, ensure_ascii=False)

Classe de validação automatizada para qualidade de dados — verifica nulos, duplicatas e gera relatórios de anomalias em JSON.

medidas_comissionamento.dax 
// Medidas DAX para análise de comissionamento
// Usadas nos dashboards de acompanhamento gerencial

% Atingimento Meta =
VAR _realizado =
    SUM( fato_producao[vl_premio] )
VAR _meta =
    SUM( dim_meta[vl_meta] )
RETURN
    DIVIDE( _realizado, _meta, 0 )

// Comissão acumulada com filtro de contexto
Comissão YTD =
CALCULATE(
    SUM( fato_comissao[vl_comissao] ),
    DATESYTD( dim_calendario[dt_referencia] )
)

// Ranking de corretores por produção
Ranking Corretor =
RANKX(
    ALL( dim_corretor[nome_corretor] ),
    [Total Prêmio],
    ,
    DESC,
    Dense
)

// Variação mês a mês
Variação MoM =
VAR _atual = [Total Prêmio]
VAR _anterior =
    CALCULATE(
        [Total Prêmio],
        DATEADD( dim_calendario[dt_referencia], -1, MONTH )
    )
RETURN
    DIVIDE( _atual - _anterior, _anterior, 0 )
